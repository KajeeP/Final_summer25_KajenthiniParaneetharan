CREATE TABLE ApplicantTable (
    applicant_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    full_name VARCHAR2(50),
    contact_number VARCHAR2(10),
    education_level VARCHAR2(30),
    application_date DATE,
    salary NUMBER(10,2)
);

CREATE TABLE EmploymentTable (
    employment_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    applicant_id INT,
    status VARCHAR(30),
    FOREIGN KEY (applicant_id) REFERENCES ApplicantTable(id)
    );




CREATE TABLE EmploymentApplication (
    application_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    applicant_id NUMBER NOT NULL,
    position_applied VARCHAR2(50),
    application_status VARCHAR2(30),
    FOREIGN KEY (applicant_id) REFERENCES ApplicantTable(id)
);
